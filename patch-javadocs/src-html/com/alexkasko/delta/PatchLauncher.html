001    package com.alexkasko.delta;
002    
003    import org.apache.commons.cli.*;
004    
005    import java.io.File;
006    
007    import static java.lang.System.out;
008    
009    /**
010     * Delta patch launcher class
011     *
012     * @author alexkasko
013     * Date: 11/19/11
014     */
015    public class PatchLauncher {
016    
017        private static final String HELP_OPTION = "help";
018    
019        /**
020         * app entry point
021         */
022        public static void main(String[] args) throws Exception {
023            Options options = new Options();
024            try {
025                options.addOption("h", HELP_OPTION, false, "show this page");
026                CommandLine cline = new GnuParser().parse(options, args);
027                String[] argList = cline.getArgs();
028                if (cline.hasOption(HELP_OPTION)) {
029                    throw new ParseException("Printing help page:");
030                } else if(2 == argList.length) {
031                    new DirDeltaPatcher().patch(new File(argList[0]), new File(argList[1]));
032                } else {
033                    throw new ParseException("Incorrect arguments received!");
034                }
035            } catch (ParseException e) {
036                HelpFormatter formatter = new HelpFormatter();
037                out.println(e.getMessage());
038                formatter.printHelp("java -jar delta-patch.jar dir patch.zip", options);
039            }
040        }
041    }
