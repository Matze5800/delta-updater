029    /**
030     * Creates ZIP file (or stream) with GDIFF deltas for all changed files and '.index' text file
031     * (with '.index_' prefix) with list of unchanged, added, updated and deleted files with SHA1 hash sums
032     *
033     * @author alexkasko
034     * Date: 11/18/11
035     */
036    public class DirDeltaCreator {
037        private static final String EMPTY_STRING = "";
038    
039        /**
040         * Creates patch ZIP file
041         *
042         * @param oldDir old version of directory
043         * @param newDir new version of directory
044         * @param outFile file to write patch into
045         * @throws IOException on any io or consistency problem
046         */
047        public void create(File oldDir, File newDir, File outFile) throws IOException {
048            OutputStream out = null;
049            try {
050                out = FileUtils.openOutputStream(outFile);
051                IOFileFilter filter = TrueFileFilter.TRUE;
052                create(oldDir, newDir, filter, out);
053            } finally {
054                IOUtils.closeQuietly(out);
055            }
056        }
057    
058        /**
059         * Writes zipped patch into provided output stream
060         *
061         * @param oldDir old version of directory
062         * @param newDir new version of directory
063         * @param filter IO filter to select files
064         * @param patch output stream to write patch into
065         * @throws IOException on any io or consistency problem
066         */
067        public void create(File oldDir, File newDir, IOFileFilter filter, OutputStream patch) throws IOException {
068            DeltaIndex paths = readDeltaPaths(oldDir, newDir, filter);
069            ZipOutputStream out = new ZipOutputStream(new BufferedOutputStream(patch));
070            writeIndex(paths, out);
071            writeCreated(paths.created, newDir, out);
072            writeUpdated(paths.updated, oldDir, newDir, out);
073            out.close();
074        }
075    
076        private DeltaIndex readDeltaPaths(File oldDir, File newDir, IOFileFilter filter) throws IOException {
077            if(!(null != oldDir && oldDir.exists() && oldDir.isDirectory())) throw new IOException("Bad oldDir argument");
078            if(!(null != newDir && newDir.exists() && newDir.isDirectory())) throw new IOException("Bad newDir argument");
079            // read files
080            Collection<File> oldFiles = listFiles(oldDir, filter, filter);
081            Collection<File> newFiles = listFiles(newDir, filter, filter);
082            // want to do comparing on strings, without touching FS
083            Set<String> oldSet = ImmutableSet.copyOf(Collections2.transform(oldFiles, new Relativiser(oldDir)));
084            Set<String> newSet = ImmutableSet.copyOf(Collections2.transform(newFiles, new Relativiser(newDir)));
085            // partitioning
086            List<String> createdPaths = Ordering.natural().immutableSortedCopy(Sets.difference(newSet, oldSet));
087            List<String> existedPaths = Ordering.natural().immutableSortedCopy(Sets.intersection(oldSet, newSet));
088            List<String> deletedPaths = Ordering.natural().immutableSortedCopy(Sets.difference(oldSet, newSet));
089            // converting
090            ImmutableList<IndexEntry.Created> created = ImmutableList.copyOf(Lists.transform(createdPaths, new CreatedIndexer(newDir)));
091            ImmutableList<IndexEntry.Deleted> deleted = ImmutableList.copyOf(Lists.transform(deletedPaths, new DeletedIndexer(oldDir)));
092            List<IndexEntry> existed = Lists.transform(existedPaths, new ExistedIndexer(oldDir, newDir));
093            // partitioning
094            ImmutableList<IndexEntry.Updated> updated = ImmutableList.copyOf(Iterables.filter(existed, IndexEntry.Updated.class));
095            ImmutableList<IndexEntry.Unchanged> unchanged = ImmutableList.copyOf(Iterables.filter(existed, IndexEntry.Unchanged.class));
096            return new DeltaIndex(created, deleted, updated, unchanged);
097        }
098    
099        private void writeIndex(DeltaIndex paths, ZipOutputStream out) throws IOException {
100            // lazy transforms
101            out.putNextEntry(new ZipEntry(".index_" + UUID.randomUUID().toString()));
102            Gson gson = new GsonBuilder().create();
103            Writer writer = new OutputStreamWriter(out, Charset.forName("UTF-8"));
104            for (IndexEntry ie : paths.getAll()) {
105                gson.toJson(ie, IndexEntry.class, writer);
106                writer.write("\n");
107            }
108            writer.flush();
109            out.closeEntry();
110        }
111    
112        private void writeCreated(List<IndexEntry.Created> paths, File newDir, ZipOutputStream out) throws IOException {
113            for(IndexEntry.Created en : paths) {
114                out.putNextEntry(new ZipEntry(en.path));
115                File file = new File(newDir, en.path);
116                FileUtils.copyFile(file, out);
117                out.closeEntry();
118            }
119        }
120    
121        private void writeUpdated(List<IndexEntry.Updated> paths, File oldDir, File newDir, ZipOutputStream out) throws IOException {
122            for(IndexEntry.Updated en : paths) {
123                out.putNextEntry(new ZipEntry(en.path + ".gdiff"));
124                File source = new File(oldDir, en.path);
125                File target = new File(newDir, en.path);
126                computeDelta(source, target, out);
127                out.closeEntry();
128            }
129        }
130    
131        private void computeDelta(File source, File target, OutputStream out) throws IOException {
132            OutputStream guarded = new NoCloseOutputStream(out);
133            GDiffWriter writer = new GDiffWriter(guarded);
134            new Delta().compute(source, target, writer);
135        }
136    
137        private static class Relativiser implements Function<File, String> {
138            private final String parent;
139    
140            private Relativiser(File parent) {
141                this.parent = separatorsToUnix(parent.getPath());
142            }
143    
144            @Override
145            public String apply(File input) {
146                String path = separatorsToUnix(input.getPath());
147                // check whether actual child
148                checkArgument(parent.equals(path.substring(0, parent.length())));
149                String relative = path.substring(parent.length() + 1);
150                return IOCase.SYSTEM.isCaseSensitive() ? relative : relative.toLowerCase();
151            }
152        }
153    
154        private static class CreatedIndexer implements Function<String, IndexEntry.Created> {
155            private final File parent;
156    
157            private CreatedIndexer(File parent) {
158                this.parent = parent;
159            }
160    
161            @Override
162            public IndexEntry.Created apply(String path) {
163                String sha1 = computeSha1(new File(parent, path));
164                return new IndexEntry.Created(path, EMPTY_STRING, sha1);
165            }
166        }
167    
168        private static class DeletedIndexer implements Function<String, IndexEntry.Deleted> {
169            private final File parent;
170    
171            private DeletedIndexer(File parent) {
172                this.parent = parent;
173            }
174    
175            @Override
176            public IndexEntry.Deleted apply(String path) {
177                String sha1 = computeSha1(new File(parent, path));
178                return new IndexEntry.Deleted(path, sha1, EMPTY_STRING);
179            }
180        }
181    
182        private static class ExistedIndexer implements Function<String, IndexEntry> {
183            private final File oldParent;
184            private final File newParent;
185    
186            private ExistedIndexer(File oldParent, File newParent) {
187                this.oldParent = oldParent;
188                this.newParent = newParent;
189            }
190    
191            @Override
192            public IndexEntry apply(String path) {
193                String oldSha1 = computeSha1(new File(oldParent, path));
194                String newSha1 = computeSha1(new File(newParent, path));
195                if(oldSha1.equals(newSha1)) {
196                    return new IndexEntry.Unchanged(path, oldSha1, newSha1);
197                } else {
198                    return new IndexEntry.Updated(path, oldSha1, newSha1);
199                }
200            }
201        }
